#Iniciar API
yarn dev

# MIGRATIONS
yarn sequelize migration:create --name=NOME_DA_MIGRATIONS
yarn sequelize db:migrate             -executa migrations pendentes
yarn sequelize db:migrate:undo        -desfaz ultima migration
yarn sequelize db:migrate:undo:all    -desfaz todas migration
